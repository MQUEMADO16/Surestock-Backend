com.surestock
│
├── SurestockApplication.java        // [Main] Entry point
│
├── config/                          // [Configuration]
│   ├── SecurityConfig.java          // Security setup (CSRF disabled, public API)
│   └── WebConfig.java               // CORS setup (Allows React to connect)
│
├── controller/                      // [API Layer] REST Endpoints
│   ├── AuthController.java          // Login/Register
│   ├── ProductController.java       // Product & Inventory management
│   └── ReportController.java        // Analytics endpoints
│
├── dto/                             // [Data Transfer Objects] JSON shapes
│   ├── auth/
│   │   └── RegisterRequestDTO.java  // Clean DTO for registration
│   ├── report/
│   │   └── ReportResultDTO.java     // JSON response for reports
│   └── ProductDTO.java              // JSON for creating products
│
├── model/                           // [Database Entities] Tables
│   ├── Product.java                 // (Product + Inventory columns)
│   ├── Role.java                    // (Enum) OWNER, EMPLOYEE
│   ├── SalesTransaction.java
│   └── User.java
│
├── repository/                      // [Data Layer] Database Access
│   ├── ProductRepository.java
│   ├── TransactionRepository.java
│   └── UserRepository.java
│
└── service/                         // [Logic Layer] Business Rules
    ├── ProductService.java
    ├── ReportService.java           // (Context) Strategy selector
    ├── UserService.java
    │
    └── report/                      // [Strategy Pattern]
        ├── InventoryReportStrategy.java // (Concrete)
        ├── IReportStrategy.java     // (Interface)
        └── SalesReportStrategy.java // (Concrete)
